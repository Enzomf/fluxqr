   'use server'
   import { createClient } from '@/lib/supabase/server'
   import { redirect } from 'next/navigation'

   export async function signInWithGoogle() {
     const supabase = await createClient()
     const { data, error } = await supabase.auth.signInWithOAuth({
       provider: 'google',
       options: {
         redirectTo: `${process.env.NEXT_PUBLIC_SITE_URL}/auth/callback`,
       },
     })
     if (data.url) redirect(data.url)
   }

1. **Create src/app/auth/callback/route.ts** following RESEARCH.md Pattern 5 exactly:
   - Export async function `GET(request: Request)`
   - Extract `code` from URL search params
   - If code exists: create a Supabase server client (using `createServerClient` directly from `@supabase/ssr` with `cookies()` — same getAll/setAll pattern as server.ts), call `exchangeCodeForSession(code)`
   - Redirect to `${origin}/dashboard` regardless (even if no code, user lands on dashboard which will redirect to login if no session)
   - IMPORTANT: This route handler creates its own Supabase client inline (not importing from server.ts) because it needs the specific cookie handling for route handlers. Follow the exact pattern from RESEARCH.md.

2. **Create src/proxy.ts** following RESEARCH.md Pattern 4:
   - Export async function `proxy(request: NextRequest)` — NOT `middleware`, the function name MUST be `proxy` for Next.js 16
   - Create Supabase server client with the **double cookie write pattern**:
     - `getAll()` reads from `request.cookies`
     - `setAll()` writes to BOTH `request.cookies` (for downstream Server Components) AND `supabaseResponse.cookies` (for browser)
   - Use `getClaims()` for auth check (preferred in Next.js 16 for performance — local JWT verification). If `getClaims()` is not available on the installed @supabase/ssr version, fall back to `getUser()`.
   - If no claims/user AND pathname starts with `/dashboard`: redirect to `/login`
   - Export `config` with matcher that excludes static files:
     ```typescript
     export const config = {
       matcher: [
         '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
       ],
     }
     ```

   CRITICAL anti-patterns to avoid:
   - Do NOT name the file `middleware.ts` — it's deprecated in Next.js 16
   - Do NOT export function named `middleware` — export `proxy`
   - Do NOT use individual cookie methods (get/set/remove) — only getAll/setAll
   - Do NOT skip the request.cookies.set step in setAll — this is essential for Server Components to see refreshed tokens
  </action>
